import React, { useState, useEffect } from 'react';
import NotificationService from './services/NotificationService';
import NotificationBell from './components/NotificationBell';
import checkHealthMetricsAndAlert from './utils/HealthMetricsAlert';

// Add this inside your main component
export default function YourMainComponent() {
  const [selectedUser, setSelectedUser] = useState(null);
  const [notificationsEnabled, setNotificationsEnabled] = useState(false);

  // Initialize notifications on app load
  useEffect(() => {
    const initNotifications = async () => {
      const granted = await NotificationService.requestPermission();
      setNotificationsEnabled(granted);
      if (granted) {
        console.log('✅ Notifications enabled for this browser');
      }
    };
    initNotifications();
  }, []);

  // Start reminder checks when user is selected
  useEffect(() => {
    if (selectedUser && notificationsEnabled) {
      NotificationService.startReminderCheck(selectedUser.id);
      console.log('✅ Started reminder checks for user:', selectedUser.name);
    } else if (!selectedUser) {
      NotificationService.stopReminderCheck();
    }
    
    return () => {
      if (selectedUser) {
        NotificationService.stopReminderCheck();
      }
    };
  }, [selectedUser, notificationsEnabled]);

  // Modified saveHealthData function to include alerts
  const saveHealthDataWithAlerts = async (healthData) => {
    try {
      // Save health data to backend (your existing code)
      const response = await fetch('http://localhost:8000/api/health-data', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(healthData)
      });
      
      if (response.ok && selectedUser) {
        // Check for health alerts
        const alerts = await checkHealthMetricsAndAlert(selectedUser.id, healthData);
        if (alerts.length > 0) {
          console.log(⚠️  health alerts triggered:, alerts);
          // Show in-app notification
          alert(Health Alerts:\n);
        }
      }
    } catch (error) {
      console.error('Failed to save health data:', error);
    }
  };

  // Add the notification bell to your header/UI
  return (
    <div>
      <header className="flex justify-between items-center p-4 bg-white dark:bg-gray-800 shadow">
        <h1 className="text-xl font-bold">Medical Health Reminder Tracker</h1>
        <div className="flex items-center space-x-4">
          {selectedUser && <NotificationBell userId={selectedUser.id} />}
          {/* Your other header content */}
        </div>
      </header>
      
      {/* Rest of your app content */}
    </div>
  );
}
